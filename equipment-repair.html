⚙ Google Sheets Setup ▾
  1. Open Google Sheets → create a new sheet
  2. Add headers: Timestamp Equipment ID Name Issue Urgency Photo
  3. Go to Extensions → Apps Script
  4. Paste the Apps Script code below and click Deploy → New deployment → Web App → "Anyone"
  5. Copy the deployment URL and paste it here:

Apps Script code to paste:

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([
    data.timestamp, data.equipmentId,
    data.equipmentName, data.issue,
    data.urgency, data.photo || ""
  ]);
  return ContentService.createTextOutput(
    JSON.stringify({result:"success"})
  ).setMimeType(ContentService.MimeType.JSON);
}
